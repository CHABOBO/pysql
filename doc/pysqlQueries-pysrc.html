233      "waitEvents"    :    """select cpu*10 "CPU", parallel*10 "Parallel execution", filewrite*10 "DB File Write", 
234                          writecomplete*10 "Write Complete", fileread*10 "DB File Read", singleread*10

236                          log*10 "Log file", net*10 "SQL*Net",

251                              sum(decode(substr(event,1,3),'log',time_waited,0)) log, 
252                              sum(decode(substr(event,1,7),'SQL*Net',time_waited,0))
